Box-sizing:
	(Content-box){ 
		width = content
		final width => content + padding + border
		Ejemplo:
			width:300px; 
			padding:20px; x2
			border:5px solid black; x2
			final width:350px;
	}
	(Border-box){
		width = content + padding + border 
		content = width - padding - border
		Ejemplo:
			width:300px;
			padding:20px; x2
			border:5px solid black; x2
			content: 250px;
